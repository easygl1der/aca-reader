\chapter{Curves}\label{chap:curves}

微积分研究的是函数的变化，而微分几何研究的是空间中的曲线和曲面。
本章我们学习\textbf{曲线}的理论，为后续学习曲面奠定基础。

本章学习目标：
\begin{enumerate}
    \item 理解参数曲线、正则曲线、简单曲线的定义
    \item 掌握弧长的定义和弧长参数化
    \item 理解曲率和挠率的几何意义
    \item 掌握 Frenet-Serret 公式
    \item 了解平面曲线的整体性质
\end{enumerate}

% ==================== Section 1-1: Introduction ====================
\section{Introduction}\label{sec:introduction}

本章我们将学习空间曲线和平面曲线的基本理论。曲线是微分几何中最基本的对象，
我们可以把它理解为空间中一条连续运动质点的轨迹。

在微分几何中，我们关心的是曲线的几何性质——如弯曲程度、扭曲程度——而不是它如何被参数化。
这引导我们引出弧长、曲率、挠率等核心概念。

\begin{Remark}
本章主要参考 Do Carmo, \textit{Differential Geometry of Curves and Surfaces}, Chapter 1。
\end{Remark}

\begin{Definition}[参数曲线]\label{def:parametrized-curve}
设 $I \subset \mathbb{R}$ 是一个区间。映射
\[
\alpha: I \to \mathbb{R}^3
\]
称为一条\textbf{参数曲线}（parametrized curve）。如果 $\alpha$ 是可微的，我们称其为\textbf{可微曲线}（differentiable curve）。
\end{Definition}

\begin{Remark}
参数曲线可以理解为质点在空间中的运动轨迹。参数 $t$ 代表时间，$\alpha(t)$ 代表 $t$ 时刻质点的位置。
\end{Remark}

\begin{Definition}[正则曲线]\label{def:regular-curve}
如果对于所有 $t \in I$，导数 $\alpha'(t) \neq 0$，则称 $\alpha$ 为\textbf{正则曲线}（regular curve）。
这意味着曲线在每一点都有切向量，且切向量不为零。
\end{Definition}

\begin{Remark}
正则性条件保证曲线是光滑的，没有尖点或结点。
\end{Remark}

\begin{Definition}[简单曲线]\label{def:simple-curve}
如果映射 $\alpha$ 是单射（即 $\alpha(t_1) = \alpha(t_2)$ 蕴含 $t_1 = t_2$），则称曲线为\textbf{简单曲线}（simple curve）。
简单曲线不自交。
\end{Definition}

\begin{Example}[圆]\label{ex:circle}
参数方程
\[
\alpha(t) = (a\cos t, a\sin t, 0), \quad t \in [0, 2\pi]
\]
表示 $\mathbb{R}^3$ 中半径为 $a$ 的圆。这是平面曲线（所有点都在 $xy$-平面上）。
\end{Example}

\begin{Example}[圆柱螺旋线]\label{ex:helix}
参数方程
\[
\alpha(t) = (a\cos t, a\sin t, bt), \quad t \in \mathbb{R}
\]
称为\textbf{圆柱螺旋线}（cylindrical helix），其中 $a > 0$ 是半径，$b \neq 0$ 是螺距参数。
圆柱螺旋线是空间曲线，它在圆柱面上缠绕。如图\ref{fig:helix}和图\ref{fig:helix-4}所示。
\end{Example}

\begin{Remark}
正则曲线要求导数非零，这保证了曲线是"光滑"的，没有尖点。
平面曲线与空间曲线的区别在于是否存在挠率：平面曲线的挠率恒为零。
\end{Remark}

给定一条可微曲线 $\alpha: [a,b] \to \mathbb{R}^3$，我们定义其弧长。

\begin{Definition}[弧长]\label{def:arc-length}
曲线 $\alpha$ 从 $t=a$ 到 $t$ 的弧长定义为
\[
s(t) = \int_a^t \|\alpha'(u)\| \, du.
\]
这里 $\|\alpha'(u)\|$ 是速度向量的模长。
\end{Definition}

\begin{Proposition}[弧长与参数化无关]\label{prop:arc-length-independence}
弧长的定义与参数化选择无关。即如果 $\beta(s)$ 是 $\alpha$ 的另一个参数化，
则从起点到对应点的弧长相同。
\end{Proposition}

\noindent\textbf{证明：} 设 $\alpha: [a,b] \to \mathbb{R}^3$ 是正则曲线，$t = t(\tau)$ 是另一个参数化，
满足 $dt/d\tau > 0$。则
\[
\frac{ds}{d\tau} = \frac{ds}{dt} \frac{dt}{d\tau} = \|\alpha'(t)\| \frac{dt}{d\tau} = \|\beta'(\tau)\|.
\]
因此从起点到任一点的弧长相同。\hfill $\square$

\begin{Definition}[单位速度曲线]\label{def:unit-speed-curve}
如果 $\|\alpha'(t)\| = 1$ 对所有 $t$ 成立，则称 $\alpha$ 为\textbf{单位速度曲线}（unit speed curve）或\textbf{弧长参数化曲线}。
\end{Definition}

\begin{Theorem}[弧长参数化]\label{th:reparametrization-by-arc-length}
每条正则曲线都可以重新参数化为单位速度曲线。
\end{Theorem}

\noindent\textbf{证明：} 设 $\alpha: I \to \mathbb{R}^3$ 是正则曲线，定义
\[
s(t) = \int_{t_0}^t \|\alpha'(u)\| \, du,
\]
其中 $t_0 \in I$ 是固定点。由于 $\alpha$ 是正则的，$s'(t) = \|\alpha'(t)\| > 0$，所以 $s$ 是单调递增的，
我们可以取其逆函数 $t(s)$。则 $\beta(s) = \alpha(t(s))$ 是单位速度曲线。\hfill $\square$

\begin{Remark}
从现在开始，除非特别说明，我们将使用弧长 $s$ 作为参数，即 $\alpha(s)$ 是单位速度曲线。
弧长是曲线的内在不变量，与参数化无关。弧长参数化后，速度向量恒为1，这是最自然的参数化方式。
\end{Remark}

% ==================== Section 1-4: The Vector Product in R3 ====================
\section{The Vector Product in R3}\label{sec:vector-product}

向量积（叉积）是研究空间曲线的重要工具。

\begin{Definition}[向量积]\label{def:cross-product}
对于 $\mathbb{R}^3$ 中的两个向量 $\mathbf{u} = (u_1, u_2, u_3)$ 和 $\mathbf{v} = (v_1, v_2, v_3)$，
它们的\textbf{向量积}（cross product）定义为
\[
\mathbf{u} \times \mathbf{v} = (u_2 v_3 - u_3 v_2, u_3 v_1 - u_1 v_3, u_1 v_2 - u_2 v_1).
\]
\end{Definition}

\begin{Remark}
向量积 $\mathbf{u} \times \mathbf{v}$ 垂直于 $\mathbf{u}$ 和 $\mathbf{v}$ 所在的平面，
其方向由右手定则确定：右手四指从 $\mathbf{u}$ 转向 $\mathbf{v}$，拇指指向 $\mathbf{u} \times \mathbf{v}$ 的方向。
\end{Remark}

\begin{Proposition}[向量积的性质]\label{prop:cross-product-properties}
设 $\mathbf{u}, \mathbf{v}, \mathbf{w} \in \mathbb{R}^3$，则：
\begin{enumerate}
\item $\mathbf{u} \times \mathbf{v} = -\mathbf{v} \times \mathbf{u}$（反交换律）
\item $(\mathbf{u} \times \mathbf{v}) \cdot \mathbf{w} = \det(\mathbf{u}, \mathbf{v}, \mathbf{w})$（标量三重积）
\item $\|\mathbf{u} \times \mathbf{v}\| = \|\mathbf{u}\| \|\mathbf{v}\| \sin\theta$，其中 $\theta$ 是 $\mathbf{u}$ 和 $\mathbf{v}$ 之间的夹角
\item $\mathbf{u} \times \mathbf{v} = \mathbf{0}$ 当且仅当 $\mathbf{u}$ 和 $\mathbf{v}$ 平行
\end{enumerate}
\end{Proposition}

\begin{Remark}
向量积在曲线理论中的应用：
\begin{itemize}
\item 构造副法向量：$\mathbf{b} = \mathbf{t} \times \mathbf{n}$
\item 计算挠率：$\tau = \dfrac{\det(\alpha', \alpha'', \alpha''')}{k^2}$
\end{itemize}
\end{Remark}

\begin{figure}[H]
\subfloat[曲率几何意义]{\label{fig:curvature}\includegraphics[width=0.8\textwidth]{figures/chapter1/fig_1-13.png}}

这是曲线理论的核心部分。我们将看到曲线的形状完全由两个函数决定：曲率（curvature）和挠率（torsion）。

\subsection{The Curvature}

\begin{Definition}[曲率]\label{def:curvature}
设 $\alpha(s)$ 是单位速度曲线。定义\textbf{曲率}（curvature）为
\[
k(s) = \|\alpha''(s)\|.
\]
即加速度向量的模长。
\end{Definition}

\begin{Remark}
曲率描述曲线弯曲的程度。
\begin{itemize}
\item 直线：曲率为 $0$，完全不弯曲
\item 圆：曲率为常数 $1/a$（$a$ 为半径），弯曲程度均匀
\item 曲率越大，曲线弯曲越厉害
\end{itemize}
\end{Remark}

\begin{Example}[圆的曲率]\label{ex:circle-curvature}
单位圆的参数方程 $\alpha(s) = (\cos s, \sin s, 0)$，则
\[
\alpha'(s) = (-\sin s, \cos s, 0), \quad \alpha''(s) = (-\cos s, -\sin s, 0),
\]
所以 $k(s) = \|\alpha''(s)\| = 1$。更一般地，半径为 $a$ 的圆的曲率为 $1/a$。
\end{Example}

\begin{Example}[直线的曲率]\label{ex:line-curvature}
直线 $\alpha(s) = p + v s$（$v$ 为单位向量），则 $\alpha''(s) = 0$，所以 $k(s) = 0$。
\end{Example}

\subsection{The Torsion (Or Bending)}

挠率描述曲线"离开"所在平面的程度。

\begin{Definition}[挠率]\label{def:torsion}
设 $\alpha(s)$ 是单位速度曲线，假设 $k(s) > 0$。定义\textbf{挠率}（torsion）为
\[
\tau(s) = \frac{\det(\alpha'(s), \alpha''(s), \alpha'''(s))}{k(s)^2}.
\]
\end{Definition}

\begin{Remark}
挠率描述曲线在空间中的扭曲程度。
\begin{itemize}
\item 平面曲线：挠率恒为零（$\det(\alpha', \alpha'', \alpha''') = 0$）
\item 空间曲线：挠率可能非零，描述曲线"扭转"的程度
\end{itemize}
\end{Remark}

\begin{Example}[圆柱螺旋线的挠率]\label{ex:helix-torsion}
圆柱螺旋线 $\alpha(t) = (a\cos t, a\sin t, bt)$ 的曲率为 $a/(a^2+b^2)$，挠率为 $b/(a^2+b^2)$，
两者均为常数。这解释了为什么螺旋线看起来均匀地"扭曲"。
\end{Example}

\begin{Remark}
曲率是曲线"弯曲"的程度，挠率是曲线"扭曲"的程度。
平面曲线：曲率可能非零，但挠率恒为零。
空间曲线：曲率和挠率都可能非零。
\end{Remark}

\subsection{The Frenet-Serret Formulas}

Frenet-Serret 公式是微分几何中最重要的公式之一，它建立了曲线的局部不变量之间的关系。

\begin{Definition}[Frenet 标架]\label{def:frenet-frame}
设 $\alpha(s)$ 是单位速度曲线，$k(s) > 0$。定义：
\begin{enumerate}
\item \textbf{切向量}（tangent vector）：$\mathbf{t}(s) = \alpha'(s)$
\item \textbf{法向量}（principal normal）：$\mathbf{n}(s) = \dfrac{\mathbf{t}'(s)}{\|\mathbf{t}'(s)\|} = \dfrac{\alpha''(s)}{k(s)}$
\item \textbf{副法向量}（binormal）：$\mathbf{b}(s) = \mathbf{t}(s) \times \mathbf{n}(s)$
\end{enumerate}
$\{\mathbf{t}, \mathbf{n}, \mathbf{b}\}$ 称为\textbf{Frenet 标架}（Frenet frame）。
\end{Definition}

\begin{Remark}
$\{\mathbf{t}, \mathbf{n}, \mathbf{b}\}$ 在每一点构成 $\mathbb{R}^3$ 的标准正交基（右手系）。
\begin{itemize}
\item $\mathbf{t}$：指向曲线前进方向
\item $\mathbf{n}$：指向曲线弯曲方向（法向量方向）
\item $\mathbf{b}$：垂直于密切平面（副法向量方向）
\end{itemize}
\end{Remark}

\begin{Definition}[密切平面]\label{def:osculating-plane}
由 $\mathbf{t}(s)$ 和 $\mathbf{n}(s)$ 张成的平面称为\textbf{密切平面}（osculating plane）。
这是曲线在该点附近最"贴切"的平面。如图\ref{fig:osculating-plane}所示。
\end{Definition}

\begin{figure}[H]
\subfloat[密切平面]{\label{fig:osculating-plane}\includegraphics[width=0.8\textwidth]{figures/chapter1/fig_1-15.png}}
\hfill
\subfloat[副法向量与平面]{\label{fig:binormal-plane}\includegraphics[width=0.8\textwidth]{figures/chapter1/fig_1-16.png}}
\end{figure}

\begin{Definition}[法平面与化直平面]\label{def:normal-rectifying-planes}
\begin{itemize}
\item \textbf{法平面}（normal plane）：由 $\mathbf{n}(s)$ 和 $\mathbf{b}(s)$ 张成的平面
\item \textbf{化直平面}（rectifying plane）：由 $\mathbf{t}(s)$ 和 $\mathbf{b}(s)$ 张成的平面
\end{itemize}
\end{Definition}

\begin{Theorem}[Frenet-Serret 公式]\label{th:frenet-serret}
设 $\alpha(s)$ 是单位速度曲线，$k(s) > 0$，挠率为 $\tau(s)$。则
\[
\begin{cases}
\mathbf{t}' = k \mathbf{n}, \\
\mathbf{n}' = -k \mathbf{t} + \tau \mathbf{b}, \\
\mathbf{b}' = -\tau \mathbf{n}.
\end{cases}
\]
\end{Theorem}

\begin{Remark}
Frenet-Serret 公式表明，如果我们知道曲率 $k(s)$ 和挠率 $\tau(s)$，以及初始的 Frenet 标架，
就可以唯一确定曲线的形状。这两个函数完全决定了曲线（在局部意义上）。
这类似于刚体运动学：给定线速度和角速度，就能确定刚体的运动。
\end{Remark}

\begin{Corollary}[曲率和挠率的计算公式]\label{cor:curvature-torsion-formula}
曲率和挠率可以通过以下公式计算：
\[
k = \|\mathbf{t}'\|, \quad \tau = -\langle \mathbf{b}', \mathbf{n} \rangle.
\]
\end{Corollary}

\begin{Theorem}[曲线存在唯一性定理]\label{th:curve-existence-uniqueness}
给定两个连续函数 $k(s) > 0$ 和 $\tau(s)$，以及 $\mathbb{R}^3$ 中一组正交标架，
存在唯一一条单位速度曲线 $\alpha(s)$，使得其曲率为 $k(s)$、挠率为 $\tau(s)$，且 Frenet 标架为给定的标架。
\end{Theorem}

\begin{Remark}
Frenet-Serret 公式是曲线的"运动方程"。曲率 $k$ 和挠率 $\tau$ 完全决定了曲线的局部几何性质。
这类似于：给定加速度和角速度，就确定了刚体的运动。
\end{Remark}

% ==================== Section 1-6: The Local Canonical Form ====================
\section{The Local Canonical Form}\label{sec:local-canonical-form}

本节研究曲线在一点附近的局部形状。由 Frenet-Serret 公式，我们可以得到曲线在弧长参数下的泰勒展开。

\begin{Theorem}[曲线在一点的局部标准形式]\label{th:local-canonical-form}
设 $\alpha(s)$ 是以弧长 $s$ 为参数的可微曲线，在 $s = s_0$ 处 $k(s_0) > 0$。则
\[
\alpha(s) = \alpha(s_0) + (s - s_0)\mathbf{t}_0 + \frac{(s - s_0)^2}{2}k_0\mathbf{n}_0 + \frac{(s - s_0)^3}{6}k_0\tau_0\mathbf{b}_0 + o((s - s_0)^3),
\]
其中 $\mathbf{t}_0 = \mathbf{t}(s_0)$，$k_0 = k(s_0)$，$\tau_0 = \tau(s_0)$。
\end{Theorem}

\begin{Remark}
这个展开式揭示了曲线的几何意义：
\begin{itemize}
\item 线性项 $(s - s_0)\mathbf{t}_0$：沿切方向 $\mathbf{t}$ 的运动
\item 二次项 $\dfrac{(s - s_0)^2}{2}k_0\mathbf{n}_0$：沿法方向 $\mathbf{n}$ 的弯曲（由曲率 $k$ 控制）
\item 三次项 $\dfrac{(s - s_0)^3}{6}k_0\tau_0\mathbf{b}_0$：沿副法方向 $\mathbf{b}$ 的扭曲（由挠率 $\tau$ 控制）
\end{itemize}
\end{Remark}

\begin{Remark}
当 $\tau(s_0) = 0$ 但 $k(s_0) \neq 0$ 时，曲线在 $s_0$ 附近是平面曲线，三次项为零。
\end{Remark}

% ==================== Section 1-7: Global Properties of Plane Curves ====================
\section{Global Properties of Plane Curves}\label{sec:global-properties-plane}

前面我们讨论的是曲线的局部性质。现在我们讨论平面曲线的整体（全局）性质。

\begin{Definition}[旋转指数]\label{def:turning-number}
设 $\alpha: [0,L] \to \mathbb{R}^2$ 是一条简单闭平面曲线，参数化为单位速度。
定义\textbf{旋转指数}（turning number）为
\[
n = \frac{1}{2\pi} \int_0^L k(s) \, ds.
\]
旋转指数表示切向量沿曲线完整一圈转过的圈数。
\end{Definition}

\begin{Theorem}[四顶点定理]\label{th:four-vertex-theorem}
设 $C$ 是平面上一条简单闭曲线，曲率恒为正。则 $C$ 上至少有四个曲率极值点（顶点）。
\end{Theorem}

\begin{Remark}
"顶点"（vertex）是指曲率取得局部极值的点。四顶点定理是平面曲线理论中最著名的整体定理之一。
例如，椭圆有四个顶点（长轴和短轴的端点）。
\end{Remark}

\begin{Theorem}[Fenchel 定理]\label{th:fenchel-theorem}
设 $\alpha: [0,L] \to \mathbb{R}^3$ 是一条简单闭曲线。则其全曲率
\[
\int_0^L k(s) \, ds \geq 2\pi,
\]
等号成立当且仅当曲线是平面上的凸曲线。
\end{Theorem}

\begin{Definition}[全曲率]\label{def:total-curvature}
曲线 $\alpha$ 的\textbf{全曲率}（total curvature）定义为
\[
\int_0^L k(s) \, ds,
\]
即曲率沿曲线的积分。
\end{Definition}

\begin{Corollary}\label{cor:fenchel-corollary}
任何简单闭曲线的全曲率至少为 $2\pi$。
\end{Corollary}

\begin{Theorem}[Fary-Milnor 定理]\label{th:fary-milnor}
如果 $\alpha$ 是一条简单闭曲线的全曲率小于 $4\pi$，则该曲线是未打结的（unknotted）。
\end{Theorem}

\begin{Remark}
四顶点定理是平面曲线的整体性质：曲线的弯曲程度必须"平衡"。
Fenchel 定理说明：闭合曲线的"总弯曲量"至少是 $2\pi$。
这些定理连接了曲线的局部几何（曲率）和全局拓扑性质。
\end{Remark}

\section{本章总结}\label{sec:chapter1-summary}

本章介绍了曲线微分几何的基础概念：

\begin{enumerate}
\item \textbf{正则曲线}：导数非零的光滑曲线
\item \textbf{弧长}：曲线的内在不变量，可用于重新参数化
\item \textbf{曲率} $k(s)$：描述曲线弯曲程度
\item \textbf{挠率} $\tau(s)$：描述曲线在空间中的扭曲程度
\item \textbf{Frenet-Serret 公式}：连接 $k$、$\tau$ 与曲线的运动
\item \textbf{局部标准形式}：曲线在一点附近的展开
\item \textbf{整体性质}：四顶点定理、Fenchel 定理、Fary-Milnor 定理
\end{enumerate}
