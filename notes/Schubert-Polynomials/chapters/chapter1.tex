\chapter{Graham Positivity of Triple Schubert Calculus}
\label{chap:TriplePositivity}

\section{预备知识}
\label{sec:Preliminaries}

在本节中，我们介绍本文所需的基本定义.

\subsection{旗流形}

\begin{Definition}[旗流形]
\label{def:FlagVariety}
设 $V$ 为 $n$ 维复向量空间. 旗流形 $Fl_n(\mathbb{C})$ 是 $V$ 中所有 flags 的集合，其中一个 flag 是如下的一系列子空间：
\begin{equation}
0 = V_0 \subset V_1 \subset V_2 \subset \cdots \subset V_n = V
\end{equation}
其中 $\dim V_i = i$. 旗流形是一个光滑的复代数簇，维度为 $\binom{n}{2}$.
\end{Definition}

旗流形是 Schubert 理论的舞台. Schubert 胞腔是旗流形的重要子簇.

\subsection{对称群与排列}

\begin{Definition}[对称群]
\label{def:SymmetricGroup}
$n$ 阶对称群 $S_n$ 是 $\{1, 2, \ldots, n\}$ 的所有置换构成的群. 一个置换 $w$ 可以记为一维数组 $w = (w(1), w(2), \ldots, w(n))$.

排列 $w$ 的逆序数 $\ell(w)$ 定义为满足 $1 \leq i < j \leq n$ 且 $w(i) > w(j)$ 的对 $(i, j)$ 的个数，也称为 $w$ 的长度.
\end{Definition}

\begin{Definition}[Schubert 多项式]
\label{def:SchubertPolynomial}
Schubert 多项式 $\{S_w(x_1, \ldots, x_n) \mid w \in S_n\}$ 是旗流形上 Schubert 类的多项式代表元，由 Lascoux 和 Schützenberger 引入. 它们满足：
\begin{equation}
S_w(x_1, \ldots, x_n) = x_1^{c_1} x_2^{c_2} \cdots + \text{更低次的项}
\end{equation}
其中 $c_i = \#\{j > i : w(j) < w(i)\}$.
\end{Definition}

\subsection{双重 Schubert 多项式}

\begin{Definition}[双重 Schubert 多项式]
\label{def:DoubleSchubertPolynomial}
双重 Schubert 多项式 $S_w(x; y)$ 是 Schubert 多项式的双变量推广，其中引入了第二组变量 $y = (y_1, \ldots, y_n)$. 它们定义为：
\begin{equation}
S_w(x; y) = \sum_{u \leq w} (-1)^{\ell(w)-\ell(u)} \mathfrak{S}_u(x) \cdot \prod_{i=1}^n (x_i - y_{w(i)})
\end{equation}
其中 $\mathfrak{S}_u(x)$ 是 Schubert 多项式，求和取遍所有满足 $u \leq w$（Bruhat 序）的置换.

双重 Schubert 多项式满足：
\begin{itemize}
\item $S_{w_0}(x; y) = \prod_{i=1}^n (x_i - y_i)$，其中 $w_0$ 是最长置换
\item 当 $y = 0$ 时，$S_w(x; 0) = S_w(x)$ 退化为经典 Schubert 多项式
\end{itemize}
\end{Definition}

\subsection{等变量子上同调}

\begin{Definition}[等变量子上同调]
\label{def:EquivariantCohomology}
设 $T \cong (\mathbb{C}^\times)^n$ 为一个 torus. 空间 $X$ 上的 $T$-等变量子上同调定义为：
\begin{equation}
H_T^*(X) = H^*(X \times_T ET)
\end{equation}
其中 $ET$ 是 $T$ 的分类空间.

对于旗流形 $Fl_n(\mathbb{C})$，其 $T$-等变量子上同调环 $H_T^*(Fl_n(\mathbb{C}))$ 具有由 Schubert 类生成的自由 $\mathbb{Z}[y_1, \ldots, y_n]$-模结构.
\end{Definition}

等变量子上同调将传统的上同调与 torus 的作用相结合，产生了丰富的代数结构.

\subsection{Grassmannian 与 Grassmannian 排列}

\begin{Definition}[Grassmannian]
\label{def:Grassmannian}
Grassmannian $Gr(k, n)$ 是 $n$ 维复向量空间中所有 $k$ 维子空间的集合. 它是旗流形的特例，满足 $V_1 \subset V_2 \subset \cdots \subset V_k$ 的 flags.
\end{Definition}

\begin{Definition}[Grassmannian 排列]
\label{def:GrassmannianPermutation}
称置换 $w \in S_n$ 为 $k$-Grassmannian 排列，如果 $w$ 的逆序数满足：
\begin{equation}
w(1) < w(2) < \cdots < w(k) \quad \text{且} \quad w(k+1) > w(k+2) > \cdots > w(n)
\end{equation}
即前 $k$ 个位置递增，后 $n-k$ 个位置递减.

$k$-Grassmannian 排列与 $Gr(k, n)$ 中的 Schubert 胞腔一一对应.
\end{Definition}

Grassmannian 排列是 Graham 原始 positivity 定理研究的对象.

\subsection{除差算子}

\begin{Definition}[除差算子]
\label{def:DividedDifference}
对于 $i = 1, \ldots, n-1$，除差算子 $\partial_i$ 定义为：
\begin{equation}
\partial_i(f) = \frac{f(x_1, \ldots, x_i, x_{i+1}, \ldots, x_n) - f(x_1, \ldots, x_{i+1}, x_i, \ldots, x_n)}{x_i - x_{i+1}}
\end{equation}
更一般地，对于置换 $w$，定义其对应的除差算子 $\partial_w$ 为适当组合的 $\partial_i$.
\end{Definition}

\begin{Definition}[Skew 除差算子]
\label{def:SkewDividedDifference}
对于两个置换 $w, v \in S_n$，Skew 除差算子 $\partial_{w/v}$ 定义为：
\begin{equation}
\partial_{w/v} = \partial_{w} \circ \partial_{v}^{-1}
\end{equation}
其中 $\partial_{v}^{-1}$ 是 $\partial_v$ 的逆算子，$\circ$ 表示算子复合（详见\cref{app:qa}中的\cref{sec:InverseDividedDifference}）。

关于除差算子和 Skew 除差算子的更多例子（详见\cref{app:qa}中的\cref{sec:SkewVsDivided}和\cref{sec:InverseDividedDifference}）。

Skew 除差算子满足以下性质：
\begin{itemize}
\item 当 $v = e$（单位置换）时，$\partial_{w/e} = \partial_w$
\item $\partial_{w/v}$ 可以展开为一系列基本除差算子的组合
\end{itemize}
\end{Definition}

\section{引言}
\label{sec:Introduction}
\label{sec:Intro}

我们在\cref{def:DoubleSchubertPolynomial}中已经定义了\textbf{双重 Schubert 多项式} $\{S_w(x; y) \mid w \in S_n\}$，它是\textbf{旗流形} $Fl_n(\mathbb{C})$ 的\textbf{等变量子上同调}类 $\{[\mathcal{X}(w)]_T\}$ 的 Lascoux-Schützenberger 型代表元（详见\cref{app:qa}中的\cref{sec:EquivariantCohomologyClass}和\cref{sec:LascouxSchutzenberger}）.+ 其中 $x = (x_1, \ldots, x_n)$ 和 $y = (y_1, \ldots, y_n)$ 是两组变量序列.

我们研究以下展开系数：
\begin{equation}
S_u(x; y) \cdot S_v(x; t) = \sum_{w \in S_\infty} c^w_{u,v}(y, t) \cdot S_w(x; t)
\end{equation}
其中 $c^w_{u,v}(y, t)$ 是关于变量 $y$ 和 $t$ 的多项式.

当 $y = t = 0$ 时，$c^w_{u,v} := c^w_{u,v}(0, 0) \in \mathbb{N}$ 是经典的 Schubert 结构常数，与 Hilbert 第十五问题密切相关. 寻找这些系数${c^w_{u,v}}$的组合模型一直是代数组合和代数几何领域的核心问题.

当 $u$ 和 $v$ 是 $k$-\textbf{Grassmannian 排列}时，Graham 在文献 \cite{Gr} 中证明了以下结果（现称为 Graham Positivity Theorem，见本文\cref{conj:Samuel}的来源论文 \cite{GX2025} 中的 Theorem 1.1）：

\begin{Theorem}[Graham Positivity Theorem]
\label{def:GrahamPositivity}
设 $u, v \in S_n$ 分别是 $k_1$-Grassmannian 和 $k_2$-Grassmannian 排列. 则
\begin{equation}
c^w_{u,v}(y, t) = \sum_{\alpha} P_\alpha(y) \cdot Q_\alpha(t)
\end{equation}
其中 $P_\alpha(y), Q_\alpha(t)$ 是具有正系数的单项式组合.
\end{Theorem}

最近，Samuel 在文献 \cite{Sa}（亦可参考本笔记\cref{sec:LiteratureSummary}）中提出了以下猜想，将上述结果推广到三重变量集的情形：

\begin{Conjecture}[Samuel 猜想]
\label{conj:Samuel}
设 $u, v \in S_\infty$ 满足一定条件（descents 条件），则
\begin{equation}
S_u(x; y) \cdot S_v(x; t) = \sum_{w \in S_\infty} c^w_{u,v}(y, t) \cdot S_w(x; t)
\end{equation}
其中 $c^w_{u,v}(y, t)$ 可以写成 $y$ 和 $t$ 的正系数多项式.
\end{Conjecture}

本文证明了上述猜想（\cref{conj:Samuel}），作为推论，还证明了 Kirillov 关于 skew 除差算子的正性猜想（\cref{def:SkewDividedDifference}）.

\section{主要定理的证明}
\label{sec:ProofMain}

本节我们介绍本文主要定理的证明思路.

\subsection{Refined Graham Positivity 的证明}
\label{sec:RefinedGrahamProof}

本文建立了一个 refined 版本的 Graham positivity 定理. 原始的 Graham positivity 定理（见\cref{def:GrahamPositivity}）要求 $u$ 和 $v$ 都是 Grassmannian 排列. 本文将其推广到更一般的情形.

\begin{Theorem}[Refined Graham Positivity {(\cite{GX2025} Theorem 1.1)}]
\label{def:RefinedGraham}
设 $u, v \in S_\infty$ 满足以下条件：存在整数 $p$ 使得
\begin{itemize}
\item 对所有 $i < p$，有 $\ell(u s_i) > \ell(u)$
\item 对所有 $i > p$，有 $\ell(v s_i) > \ell(v)$
\end{itemize}
则展开系数 $c^w_{u,v}(y, t)$ 可以写成 $y$ 和 $t$ 的正系数多项式.
\end{Theorem}

该条件被称为 \textbf{descents 条件}，它包含了原始的 Grassmannian 情形作为特例.

\textbf{证明：} \cref{def:RefinedGraham} 的证明基于 \cref{def:Lemma21}、\cref{def:Lemma22} 和 \cref{def:Lemma23}.

\begin{Lemma}[Lemma 2.1]
\label{def:Lemma21}
对于满足 descents 条件的排列 $u, v$，存在非负整数 $a_{u,v}$ 使得
$$c^w_{u,v}(y, t) = \sum_{\alpha} a_{u,v}^\alpha \cdot p_\alpha(y) \cdot q_\alpha(t)$$
其中 $p_\alpha(y), q_\alpha(t)$ 是某些单项式.
\end{Lemma}

\begin{Lemma}[Lemma 2.2]
\label{def:Lemma22}
在上述条件下，系数 $a_{u,v}^\alpha$ 是非负整数.
\end{Lemma}

\begin{Lemma}[Lemma 2.3]
\label{def:Lemma23}
$p_\alpha(y)$ 和 $q_\alpha(t)$ 可以选取为正系数的单项式.
\end{Lemma}

由\cref{def:Lemma21}、\cref{def:Lemma22}和\cref{def:Lemma23}，我们完成\cref{def:RefinedGraham}的证明.

\subsection{Billey 公式的正性几何解释}
\label{sec:BilleyPositivity}

Billey 公式是计算等变量 Schubert 系数的重要工具. 本文给出了 Billey 公式中正性的新的几何解释.

Billey 公式表明，展开系数 $c^w_{u,v}(y)$ 可以通过特定的 Schubert 胞腔的几何性质来计算. 本文证明了这些胞腔的交集具有某种正性结构，从而解释了系数的正性来源.

具体而言，对于给定的排列 $u, v, w$，Billey 公式给出：
$$c^w_{u,v}(y) = \sum_{b \in B(u,v,w)} \prod_{i=1}^{\ell(w)} y_i^{m_i(b)}$$
其中 $B(u,v,w)$ 是某个特定的胞腔集合，$m_i(b)$ 是非负整数. 本文给出了这一几何解释的新证明.

\subsection{Theorem 1.2 的证明：应用到 Triple Schubert Calculus}
\label{sec:TripleSchubertProof}

本文的第二个主要结果是证明了 Samuel 猜想（\cref{conj:Samuel}），即在三重变量集情形下的 Graham positivity.

\begin{Theorem}[Theorem 1.2 - Samuel 猜想]
设 $u, v \in S_\infty$ 满足 descents 条件，则
$$S_u(x; y) \cdot S_v(x; t) = \sum_w c^w_{u,v}(y, t) \cdot S_w(x; t)$$
其中 $c^w_{u,v}(y, t)$ 是 $y$ 和 $t$ 的正系数多项式.
\end{Theorem}

\textbf{证明：} Theorem 1.2 的证明基于 Theorem 1.1 和以下代数操作.

首先，对于双重 Schubert 多项式，我们有：
$$S_u(x; y) = \sum_{u' \leq u} c^{u'}_u(y) \cdot S_{u'}(x)$$
其中 $c^{u'}_u(y)$ 是 $y$ 的某个多项式.

类似地，
$$S_v(x; t) = \sum_{v' \leq v} c^{v'}_v(t) \cdot S_{v'}(x)$$

将两式相乘并展开，我们得到：
$$S_u(x; y) \cdot S_v(x; t) = \sum_{u' \leq u, v' \leq v} c^{u'}_u(y) \cdot c^{v'}_v(t) \cdot S_{u'}(x) \cdot S_{v'}(x)$$

应用 Theorem 1.1 于乘积 $S_{u'}(x) \cdot S_{v'}(x)$，我们得到：
$$S_{u'}(x) \cdot S_{v'}(x) = \sum_w c^w_{u',v'}(0, t) \cdot S_w(x; t)$$

结合以上各式，我们得到：
$$S_u(x; y) \cdot S_v(x; t) = \sum_w \left( \sum_{u' \leq u, v' \leq v} c^{u'}_u(y) \cdot c^{v'}_v(t) \cdot c^w_{u',v'}(0, t) \right) \cdot S_w(x; t)$$

注意到 $c^{u'}_u(y)$ 和 $c^{v'}_v(t)$ 是 $y$ 和 $t$ 的正系数多项式（由双重 Schubert 多项式的定义可直接验证），而 $c^w_{u',v'}(0, t)$ 由 Theorem 1.1 是 $t$ 的正系数多项式. 因此内层求和给出的是 $y$ 和 $t$ 的正系数多项式.

这就完成了 Theorem 1.2 的证明.

\subsection{Theorem 1.3 的证明：Grothendieck 多项式的 Positivity 猜想}
\label{sec:GrothendieckProof}

作为应用，我们证明了 Kirillov 关于 skew 除差算子的正性猜想.

\begin{Theorem}[Theorem 1.3 - Kirillov 猜想]
对于任意置换 $w$，skew 除差算子 $\partial_{w/v}$ 作用于 Schubert 多项式时，系数是非负的.
\end{Theorem}

\textbf{证明：} 该定理是 Theorem 1.2 的直接推论.

根据 skew 除差算子的定义（见\cref{def:SkewDividedDifference}），我们有：
$$\partial_{w/v}(S_u(x)) = \sum_{w'} c^{w'}_{u,v} \cdot S_{w'}(x)$$

另一方面，由 Theorem 1.2 的特殊情形（取 $y = 0$），我们可以将系数 $c^{w'}_{u,v}$ 表示为某些正系数的组合.

具体来说，设 $u = w$ 且 $v = w_0 / w$（其中 $w_0$ 是最长置换），则：
$$\partial_{w/w_0/w}(S_{w_0}(x)) = \partial_{w/v}(S_{w_0}(x))$$

由于 $S_{w_0}(x) = x_1^{n-1} x_2^{n-2} \cdots x_{n-1}$ 是某个特定的单项式，我们可以利用 Billey 公式的展开来计算.

由 Theorem 1.1 和 Billey 公式的性质，我们可以证明所有出现的系数都是非负整数.

这就完成了 Theorem 1.3 的证明.

\textbf{证明完成}.

\begin{Conjecture}[Kirillov 猜想]
\label{conj:Kirillov}
对于任意置换 $w$，skew 除差算子 $\partial_{w/v}$ 作用于 Schubert 多项式时，系数是非负的.
\end{Conjecture}
