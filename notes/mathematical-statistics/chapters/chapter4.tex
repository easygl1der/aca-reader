\chapter{一些基本的统计推断}\label{chap:elementary-statistical-inferences}

\section*{Chapter 4 概述}

在前三章中，我们建立了描述随机变量和概率分布的理论框架。现在我们面临一个根本性问题：\textbf{给定观测数据，如何推断产生这些数据的底层分布？}

这就是统计推断的核心问题。本章将介绍两种主要的推断工具：
\begin{itemize}
\item \textbf{置信区间}（Confidence Intervals）：未知参数的合理取值范围
\item \textbf{假设检验}（Hypothesis Testing）：评估关于总体的声明是否成立
\end{itemize}

\textbf{本章路线图}：抽样与统计量 $\to$ 点估计 $\to$ 置信区间 $\to$ 假设检验 $\to$ 卡方检验 $\to$ 蒙特卡洛 $\to$ Bootstrap

\section{4.1 抽样与统计量}\label{sec:sampling-statistics-ch4}

\subsection{为什么需要抽样？}

在现实中，我们通常无法观测整个总体。例如，测量所有成年人的身高是不可能的。因此，我们从总体中抽取一个随机样本，用样本信息推断总体特征。

\textbf{核心假设}：$X_1, X_2, \ldots, X_n$ 是 iid（独立同分布）随机变量，来自某个分布 $f(x; \theta)$，其中 $\theta$ 是未知参数。

\subsection{样本均值与样本方差}

给定样本，最自然的首选统计量是\textbf{样本均值}：
\begin{equation}\label{eq:sample-mean-ch4}
\bar{X} = \frac{1}{n}\sum_{i=1}^{n} X_i.
\end{equation}

\textbf{关键性质}：
\[
\mathbb{E}\bar{X} = \mu, \quad \text{var}(\bar{X}) = \frac{\sigma^2}{n}.
\]

这意味着：
\begin{itemize}
\item $\bar{X}$ 是 $\mu$ 的\textbf{无偏估计}（$E\bar{X} = \mu$）
\item $\bar{X}$ 的方差随样本量 $n$ 增大而减小（$1/n$ 衰减）
\end{itemize}

\textbf{大样本性质}：
\begin{itemize}
\item \textbf{弱大数定律}：$\bar{X} \xrightarrow{P} \mu$（当 $n \to \infty$）
\item \textbf{中心极限定理}：$\frac{\bar{X}-\mu}{\sigma/\sqrt{n}} \xrightarrow{d} N(0,1)$
\end{itemize}

\begin remark}为什么中心极限定理重要？]
即使总体分布未知，只要样本量足够大，$\bar{X}$ 的分布也近似正态。这使得我们可以构造置信区间、进行假设检验——而不需要知道总体的具体分布。
\end remark}

\textbf{样本方差}：
\begin{equation}\label{eq:sample-variance-ch4}
S^2 = \frac{1}{n-1}\sum_{i=1}^{n} (X_i - \bar{X})^2.
\end{equation}

\textbf{注意}：分母是 $n-1$ 而非 $n$！这是为了保证 $S^2$ 是 $\sigma^2$ 的无偏估计（$E(S^2) = \sigma^2$）。

\section{4.2 点估计}\label{sec:point-estimators-ch4}

\textbf{核心问题}：给定数据，如何估计未知参数 $\theta$？

点估计提供一个单一数值作为参数的估计值。

\subsection{估计量的评判标准}

\begin{definition}[偏差]\label{def:bias-ch4}
估计量 $\hat{\theta}$ 的偏差定义为 $\text{Bias}(\hat{\theta}) = \mathbb{E}(\hat{\theta}) - \theta$。若 $\mathbb{E}(\hat{\theta}) = \theta$，则 $\hat{\theta}$ 是无偏的。
\end{definition}

\begin{definition}[均方误差]\label{def:mse-ch4}
估计量 $\hat{\theta}$ 的均方误差为 $\text{MSE}(\hat{\theta}) = \mathbb{E}[(\hat{\theta} - \theta)^2]$。对无偏估计，$\text{MSE} = \text{var}(\hat{\theta})$。
\end{definition}

\textbf{偏差-方差权衡}：一个有偏估计量可能比无偏估计量有更小的 MSE。

\begin{example}[样本均值的性质]\label{ex:sample-mean-properties-ch4}
设 $X_1, \ldots, X_n$ 来自均值为 $\mu$、方差为 $\sigma^2$ 的分布。则 $\bar{X}$ 是 $\mu$ 的无偏估计，$\text{MSE}(\bar{X}) = \sigma^2/n$。
\end{example}

\subsection{矩估计与最大似然估计}

\begin theorem}[矩估计]\label{th:method-of-moments-ch4}
样本矩 $M_k = \frac{1}{n}\sum_{i=1}^{n} X_i^k$ 依概率收敛到相应的总体矩 $E(X^k)$。矩估计通过解方程组 $\hat{\mu}_k = M_k$ 得到参数估计。
\end theorem}

\begin{example}[正态分布的 MLE]\label{ex:normal-mle-ch4}
设 $X_1, \ldots, X_n \sim N(\mu, \sigma^2)$。最大似然估计（MLE）为：
\[
\hat{\mu} = \bar{X}, \quad \hat{\sigma}^2 = \frac{1}{n}\sum_{i=1}^{n} (X_i - \bar{X})^2.
\]
注意：$\hat{\mu}$ 无偏，但 $\hat{\sigma}^2$ 有偏（除以 $n$ 而非 $n-1$）。
\end{example}

\section{4.3 置信区间}\label{sec:confidence-intervals-ch4}

\textbf{核心问题}：点估计 $\hat{\theta}$ 有多好？它离真实参数有多远？

置信区间提供了一个\textbf{区间}而非单一数值，告诉你参数可能落在什么范围内。

\begin{definition}[置信区间]\label{def:confidence-interval-ch4}
设 $L = g_1(X_1, \ldots, X_n)$ 和 $U = g_2(X_1, \ldots, X_n)$ 是两个统计量。若对所有 $\theta \in \Omega$，
\[
\mathbb{P}_\theta(L \leq \theta \leq U) = 1 - \alpha,
\]
则称随机区间 $[L, U]$ 为 $\theta$ 的 $100(1-\alpha)\%$ 置信区间。
\end{definition}

\textbf{解释}：不是"$\theta$ 以 $1-\alpha$ 的概率落在区间内"（$\theta$ 是固定常数），而是"如果重复抽样，100 次中有 $100(1-\alpha)$ 次构造的区间会包含 $\theta$"。

\begin{example}[正态均值、已知方差]\label{ex:ci-normal-known-variance-ch4}
设 $X_1, \ldots, X_n \sim N(\mu, \sigma^2)$，$\sigma^2$ 已知。$\mu$ 的 $100(1-\alpha)\%$ 置信区间为：
\[
\left[ \bar{X} - z_{\alpha/2}\frac{\sigma}{\sqrt{n}}, \quad \bar{X} + z_{\alpha/2}\frac{\sigma}{\sqrt{n}} \right],
\]
其中 $z_{\alpha/2}$ 满足 $\Phi(z_{\alpha/2}) = 1 - \alpha/2$。
\end{example}

\begin{example}[正态均值、未知方差]\label{ex:ci-normal-unknown-variance-ch4}
设 $X_1, \ldots, X_n \sim N(\mu, \sigma^2)$，$\sigma^2$ 未知。$\mu$ 的 $100(1-\alpha)\%$ 置信区间为：
\[
\left[ \bar{X} - t_{\alpha/2,n-1}\frac{S}{\sqrt{n}}, \quad \bar{X} + t_{\alpha/2,n-1}\frac{S}{\sqrt{n}} \right],
\]
其中 $t_{\alpha/2,n-1}$ 是 $t(n-1)$ 分布的上侧 $\alpha/2$ 分位数。

\textbf{注意}：用 $S$ 替代 $\sigma$，用 $t$ 替代 $z$。因为未知方差带来的不确定性，区间变宽了。
\end{example}

\begin theorem}[方差置信区间]\label{th:ci-variance-ch4}
设 $X_1, \ldots, X_n \sim N(\mu, \sigma^2)$。$\sigma^2$ 的 $100(1-\alpha)\%$ 置信区间为：
\[
\left[ \frac{(n-1)S^2}{\chi^2_{\alpha/2,n-1}}, \quad \frac{(n-1)S^2}{\chi^2_{1-\alpha/2,n-1}} \right].
\]
\end theorem}

\begin remark}直觉解释]
置信区间的宽度与 $\sqrt{n}$ 成反比——样本量越大，区间越窄，估计越精确。同时，置信水平 $1-\alpha$ 越高，区间越宽。
\end remark}

\section{4.4 顺序统计量}\label{sec:order-statistics-ch4}

\textbf{核心思想}：将样本从小到大排列：
\[
X_{(1)} \leq X_{(2)} \leq \cdots \leq X_{(n)}.
\]

$X_{(k)}$ 是第 $k$ 个顺序统计量。

\textbf{样本中位数}：
\[
\tilde{X} = \begin{cases}
X_{((n+1)/2)} & n \text{ 奇数} \\
\frac{1}{2}\left(X_{(n/2)} + X_{(n/2+1)}\right) & n \text{ 偶数}
\end{cases}
\]

\textbf{样本极差}：$R = X_{(n)} - X_{(1)}$

顺序统计量的 cdf 和 pdf：
\[
F_{X_{(k)}}(x) = \sum_{j=k}^{n} \binom{n}{j} [F(x)]^j [1-F(x)]^{n-j},
\]
\[
f_{X_{(k)}}(x) = \frac{n!}{(k-1)!(n-k)!} [F(x)]^{k-1} [1-F(x)]^{n-k} f(x).
\]

\section{4.5 假设检验}\label{sec:hypothesis-testing-ch4}

假设检验回答一个问题：\textbf{数据是否提供足够证据拒绝关于总体的某个声明？}

\textbf{核心概念}：
\begin{itemize}
\item \textbf{原假设} $H_0$：被检验的假设（通常代表"无效应"或"现状"）
\item \textbf{备择假设} $H_1$：与 $H_0$ 对立的假设
\end{itemize}

\begin{center}
\begin{tabular}{c|c|c}
 & 未拒绝 $H_0$ & 拒绝 $H_0$ \\ \hline
$H_0$ 为真 & 正确 & 第一类错误 ($\alpha$) \\
$H_0$ 为假 & 第二类错误 ($\beta$) & 正确（功效 $1-\beta$）
\end{tabular}
\end{center}

\textbf{显著性水平} $\alpha = \mathbb{P}(\text{第一类错误}) = \mathbb{P}(\text{拒绝 } H_0 \mid H_0 \text{ 为真})$。

\begin{example}[正态均值检验、已知方差]\label{ex:test-normal-known-ch4}
设 $X_1, \ldots, X_n \sim N(\mu, \sigma^2)$，$\sigma^2$ 已知。检验 $H_0: \mu = \mu_0$ vs $H_1: \mu > \mu_0$。

检验统计量：$Z = \frac{\bar{X} - \mu_0}{\sigma/\sqrt{n}}$。

在 $\alpha$ 水平下拒绝 $H_0$ 当且仅当 $Z > z_\alpha$。
\end{example}

\begin{example}[正态均值检验、未知方差]\label{ex:test-normal-unknown-ch4}
设 $X_1, \ldots, X_n \sim N(\mu, \sigma^2)$，$\sigma^2$ 未知。检验 $H_0: \mu = \mu_0$ vs $H_1: \mu \neq \mu_0$。

检验统计量：$T = \frac{\bar{X} - \mu_0}{S/\sqrt{n}}$。

在 $\alpha$ 水平下拒绝 $H_0$ 当且仅当 $|T| > t_{\alpha/2,n-1}$。
\end{example}

\begin remark}$p$ 值的直观理解]
$p$ 值是在 $H_0$ 为真的假设下，观察到比当前样本更极端结果的概率。$p$ 值越小，拒绝 $H_0$ 的证据越强。
\end remark}

\section{4.6 卡方检验}\label{sec:chi-square-tests-ch4}

卡方检验用于分析\textbf{分类数据}——即计数数据。

\subsection{拟合优度检验}\label{sub:goodness-of-fit-ch4}

\textbf{问题}：数据是否来自某个特定分布？

设 $O_1, \ldots, O_k$ 是 $k$ 个类别的观测频数，$E_i = np_{i,0}$ 是 $H_0$ 下的期望频数。

\textbf{检验统计量}：
\[
\chi^2 = \sum_{i=1}^{k} \frac{(O_i - E_i)^2}{E_i}.
\]

在 $H_0$ 下，$\chi^2 \approx \chi^2(k-1)$（大样本近似）。

拒绝 $H_0$ 当 $\chi^2 > \chi^2_{\alpha, k-1}$。

\subsection{独立性检验}\label{sub:test-independence-ch4}

对于 $r \times c$ 列联表，设 $O_{ij}$ 为单元格 $(i,j)$ 的观测频数。

在独立性假设下，$E_{ij} = \frac{O_{i\cdot} O_{\cdot j}}{n}$。

\textbf{检验统计量}：
\[
\chi^2 = \sum_{i=1}^{r}\sum_{j=1}^{c} \frac{(O_{ij} - E_{ij})^2}{E_{ij}}.
\]

在 $H_0$ 下，$\chi^2 \approx \chi^2((r-1)(c-1))$。

\section{4.7 蒙特卡洛方法}\label{sec:monte-carlo-ch4}

\textbf{核心思想}：用随机模拟来近似计算概率和积分。

\subsection{为什么需要蒙特卡洛？}

有些概率/期望无法解析计算，但可以用计算机模拟近似。大数定律保证了模拟的收敛性：

\[
\frac{1}{n}\sum_{i=1}^{n} g(X_i) \xrightarrow{P} \mathbb{E}[g(X)].
\]

\begin{example}[蒙特卡洛估计 $\pi$]\label{ex:monte-carlo-pi-ch4}
在单位正方形 $[0,1] \times [0,1]$ 中均匀随机投点，落入单位圆弧内的概率为 $\pi/4$。因此 $\hat{\pi} = 4 \times \text{(落入圆内比例)}$。
\end{example}

\subsection{接受-拒绝算法}\label{sub:accept-reject-ch4}

当目标分布 $f(x)$ 难以直接抽样时，接受-拒绝算法提供了一种间接方法。

\textbf{算法}：
\begin{enumerate}
\item 选择一个提议分布 $g(x)$，满足 $f(x) \leq Mg(x)$（对某个 $M > 1$）
\item 生成 $Y \sim g$ 和 $U \sim \text{Uniform}(0,1)$ 独立
\item 若 $U \leq f(Y)/(Mg(Y))$，接受 $X = Y$；否则拒绝，返回步骤 2
\end{enumerate}

接受概率为 $1/M$。

\section{4.8 Bootstrap 方法}\label{sec:bootstrap-ch4}

\textbf{核心问题}：估计量的抽样分布通常难以解析计算，怎么办？

Bootstrap 的答案是：\textbf{用重抽样近似}。

\textbf{步骤}：
\begin{enumerate}
\item 从原始样本中有放回地抽取 $n$ 个观测，形成一个 bootstrap 样本 $X_1^*, \ldots, X_n^*$
\item 计算 bootstrap 复制 $T^* = g(X_1^*, \ldots, X_n^*)$
\item 重复步骤 1-2 多次（如 $B = 1000$ 次），得到 $T_1^*, \ldots, T_B^*$
\item 用 $T_1^*, \ldots, T_B^*$ 的经验分布近似 $T$ 的抽样分布
\end{enumerate}

\subsection{百分位 Bootstrap 置信区间}\label{sub:percentile-bootstrap-ch4}

$\theta$ 的 $100(1-\alpha)\%$ percentile bootstrap 置信区间：
\[
\left[ T_{(\alpha B)}^*, \quad T_{((1-\alpha)B)}^* \right].
\]

\begin remark}Bootstrap 的哲学]
Bootstrap 假设"样本就像总体"——通过从样本中有放回抽样，我们模拟了如果重复实验会得到什么。这是一种非参数的近似方法，不需要对分布做任何假设。
\end remark}

\section*{本章小结}

\textbf{核心概念}：
\begin{itemize}
\item \textbf{统计量}：样本的函数，不依赖未知参数
\item \textbf{点估计}：用单一数值估计参数；无偏性、渐近性、效率是主要评判标准
\item \textbf{置信区间}：参数的合理取值范围；宽度随样本量增大而减小
\item \textbf{假设检验}：判断数据是否提供足够证据拒绝 $H_0$；第一类和第二类错误的权衡
\end{itemize}

\textbf{重要公式}：
\begin{center}
\begin{tabular}{|c|c|}
\hline
\textbf{场景} & \textbf{置信区间/检验统计量} \\ \hline
正态均值、已知 $\sigma$ & $\bar{X} \pm z_{\alpha/2}\sigma/\sqrt{n}$ \\ \hline
正态均值、未知 $\sigma$ & $\bar{X} \pm t_{\alpha/2,n-1}S/\sqrt{n}$ \\ \hline
正态方差 & $\frac{(n-1)S^2}{\chi^2_{\alpha/2,n-1}}, \frac{(n-1)S^2}{\chi^2_{1-\alpha/2,n-1}}$ \\ \hline
单样本 $Z$ 检验 & $Z = \frac{\bar{X}-\mu_0}{\sigma/\sqrt{n}}$ \\ \hline
单样本 $t$ 检验 & $T = \frac{\bar{X}-\mu_0}{S/\sqrt{n}}$ \\ \hline
拟合优度 $\chi^2$ 检验 & $\chi^2 = \sum \frac{(O-E)^2}{E} \approx \chi^2(k-1)$ \\ \hline
\end{tabular}
\end{center}
